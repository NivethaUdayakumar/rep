import gzip, mmap, os, re, shutil, tempfile

def find_regex_in_gzip_with_mmap(gz_path: str, pattern: str, flags: int = 0):
    """
    Decompresses `gz_path` to a temp file, mmaps it, and yields
    (start, end, matched_text) for every regex hit.

    `pattern` is a Unicode regex. Internally we search on bytes; matches are
    decoded back to UTF-8 for convenience.
    """
    # Compile a bytes-pattern for mmap search (UTF-8)
    # Caveat: \w, \b, etc. are ASCII-only on bytes; use explicit char classes if needed.
    bpat = re.compile(pattern.encode("utf-8"), flags)

    tmp_file = None
    try:
        # 1) Decompress to a temporary file (no full in-RAM copy)
        with gzip.open(gz_path, "rb") as gz, tempfile.NamedTemporaryFile(delete=False) as tmp:
            tmp_file = tmp.name
            shutil.copyfileobj(gz, tmp)

        # 2) mmap the decompressed file and search on bytes
        with open(tmp_file, "rb") as f, mmap.mmap(f.fileno(), 0, access=mmap.ACCESS_READ) as mm:
            for m in bpat.finditer(mm):
                start, end = m.span()
                # Decode the exact match to utf-8 for user-friendly text
                matched_text = bytes(mm[start:end]).decode("utf-8", errors="replace")
                yield (start, end, matched_text)
    finally:
        if tmp_file and os.path.exists(tmp_file):
            try:
                os.remove(tmp_file)
            except OSError:
                pass

# --- Example usage ---
# Find all lines containing “error: <anything>” across newlines:
# Use DOTALL if pattern may span newlines; add IGNORECASE if needed.
for s, e, txt in find_regex_in_gzip_with_mmap(
        "big.log.gz",
        r"error:\s.*?",
        flags=re.IGNORECASE | re.DOTALL):
    print(s, e, txt)
